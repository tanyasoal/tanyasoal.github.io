Jawaban

A = C + 4

B-5 = 2(C-5)
B-5 = 2C - 10
B = 2C - 10 + 5 = 2C - 5

A + B + C = 79
(C+4) + (2C-5) + C = 79
4C - 1 = 79
4C = 80
C = 20

A = 20 + 4 = 24

umur bima misalkan r
maka p=4 + q
r - 5 = 2(q - 5)
r = 2q -10 + 5
r = 2q - 5
p + q + r = 79
(4 + q) + q + (2q - 5) = 79
4q - 1 = 79
4q = 80
q = 80/4
q =20

p = 4 + q = 4 + 20 = 24
umur ahmad sekarang(p) = 24 tahun
