Jawaban

R = 6 cm
r = 3 cm
l = 12 cm

l² = p² - (R - r)²
12² = p² - (6 - 3)²
12² = p² - 3²
144 = p² - 9
p² = 144 + 9
p² = 153
p = √153
p = √(9 x 17)
p = 3√17 cm

Jawaban

A = 25
b = 45 - 25 = 20
n = 10

Un = a + (n - 1)b
U10 = 25 + (10 - 1)20
