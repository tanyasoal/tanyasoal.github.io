cos 2x = 1- 2sin^2 x

cos 2x - sin x = 0
1 - 2sin^2 x - sinx = 0
2sin^2 x + sin x - 1 = 0
(2sin x -1)(sin x + 1) = 0
sin x = 1/2
sin x = -1

arc sin -1 = 270
