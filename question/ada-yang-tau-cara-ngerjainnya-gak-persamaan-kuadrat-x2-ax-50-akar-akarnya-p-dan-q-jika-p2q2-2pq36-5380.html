Jawaban

X² - ax - 5 = 0
x = p atau x = q
(x-p)(x-q) = 0
x² - qx - px + pq = 0
x² - (q+p)x + pq = 0

pq = -5
q + p = a

p² + q² - 2pq = 36
(p-q)² = 36
p - q = 6
p = 6 + q


pq = -5
(6+q).q = -5
6q + q² = -5
q² + 6q + 5 = 0
(q+5)(q+1) = 0
q = -5 atau q = -1

Jika q = -5
p = 6 - 5 = 1

Jika q = -1
p = 6 - 1 = 5

Nilai a
a = q + p
a = -5 + 1
a = -4

atau
a = q + p
a = -1 + 5
a = 4

Jadi, a = -4 atau a = 4
