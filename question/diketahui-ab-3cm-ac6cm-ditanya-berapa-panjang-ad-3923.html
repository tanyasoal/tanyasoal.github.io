Jawaban

A = -2
b = 3
r = -1

(a + 4b) : (-3 - r)
= (-2 + 4(3)) : (-3 - (-1))
= (-2 + 12) : (-3 + 1)
= 10 : (-2)
= -5
(a+4b) : (-3-r)
= ((-2) +4.3) : (-3-(-1))
= ((-2)+12): (-2)
= 10:(-2)
