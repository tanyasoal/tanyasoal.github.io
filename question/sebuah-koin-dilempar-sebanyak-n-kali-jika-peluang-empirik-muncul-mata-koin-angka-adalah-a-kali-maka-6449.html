x + 4y = -1

2x - 3y = 20
2 (-4y - 1) - 3y = 20
-8y - 2 - 3y = 20
-11y = 20 + 2
y = 22/-11
y = -2

x = -4y - 1
x = -4 . (-2) - 1
x = 8 - 1
x = 7

2 . x . y = 2 . 7 . (-2)
