U7 = 8
U16 = 44

Ditanya = U20..?
jawab

U16 = a + 15b = 44
U7 = a + 6b = 8
-------------------------- -
9b = 36
b = 36/9
b = 4

a + 6b = 8
a + 6(4) = 8
a + 24 = 8
a = 8 - 24
a = - 16

maka,

U20 = a + 19b
U20 = - 16 + 19(4)
U20 = - 16 + 76
U20 = 60

U16 = 15b + a = 44

a = 8 - 6b
--------------------------------
gunakan cara substitusi
--------------------------------
15b + a = 44
15b + 8 - 6b = 44
9b = 44 - 8
9b = 36
b = 4

--------------------------------
gunakan cara substitusi
--------------------------------
a = 8 - 4b
a = 8 - 4×6
a = 8 - 24
a = -16

U20 = 80 - 20
