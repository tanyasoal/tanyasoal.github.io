η = 80% = 0,8
Vp = 2000 volt
Vs = 800 volt
is = 4 A

Ditanya :
Ip = ?

Penyelesaian :
η = Vs. Is / Vp. Ip
0,8 = 800.4 / 2000. Ip
0,8 = 3200/2000Ip

Ip = 2 A
